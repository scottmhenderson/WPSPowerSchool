<SQLQuery><textarea name="UF-0060051075">Select s.Student_Number, s.LastFirst student, s.Home_Room, sec.Room, cc.Course_Number, 
case s.Home_Room
  when sec.Room then ''
  else '&lt;'
  end
FROM Students s
JOIN cc ON cc.StudentID = s.ID and cc.Course_Number like 'MPHR%'
JOIN Sections sec on cc.SectionID = sec.ID
ORDER BY student;</textarea></SQLQuery>
